```javascript
// yjs-server/src/server.js
const WebSocket = require('ws')
const http = require('http')
const { setupWSConnection } = require('y-websocket/bin/utils')
const jwt = require('jsonwebtoken')
const Redis = require('ioredis')

const server = http.createServer()
const wss = new WebSocket.Server({ server })
const redisSub = new Redis(process.env.REDIS_URL)

// 存储连接的用户 { userId: Set<ws> }
const userConnections = new Map()

wss.on('connection', (ws, req) => {
  const url = new URL(req.url, 'http://localhost')
  const token = url.searchParams.get('token')
  const roomName = url.searchParams.get('room')

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET)
    ws.userId = decoded.userId
    ws.roomName = roomName

    // 注册连接
    if (!userConnections.has(decoded.userId)) {
      userConnections.set(decoded.userId, new Set())
    }
    userConnections.get(decoded.userId).add(ws)

    // 连接 Yjs 文档
    setupWSConnection(ws, req)

    // 订阅画布频道（评论/投票事件）
    redisSub.subscribe(`canvas:${roomName}`)

    // 订阅用户频道（通知推送）
    redisSub.subscribe(`user:${decoded.userId}`)
  } catch (err) {
    ws.close(1008, 'Unauthorized')
  }
})

// Redis 消息处理
redisSub.on('message', (channel, message) => {
  if (channel.startsWith('canvas:')) {
    // 广播给同房间用户
    const canvasId = channel.replace('canvas:', '')
    wss.clients.forEach(client => {
      if (client.roomName === canvasId && client.readyState === WebSocket.OPEN) {
        client.send(message)
      }
    })
  } else if (channel.startsWith('user:')) {
    // 推送给特定用户
    const userId = channel.replace('user:', '')
    const connections = userConnections.get(userId)
    if (connections) {
      connections.forEach(ws => {
        if (ws.readyState === WebSocket.OPEN) {
          ws.send(message)
        }
      })
    }
  }
})

server.listen(3001)
```

## 风险与应对

| 风险 | 影响 | 应对措施 |
|------|------|----------|
| Docker 不熟悉 | 部署延迟 | 提前学习，准备详细文档 |
| Yjs 协作问题 | 核心功能不可用 | 使用成熟方案，多测试 |
| Excalidraw 定制难 | 媒体功能受限 | 使用覆盖层方案 |
| 性能问题 | 多人协作卡顿 | 预留优化时间，准备缓存方案 |
| 结构化工具复杂 | 思维导图/看板/泳道图开发耗时 | 使用 JSONB 存储，减少表结构复杂度 |
| Redis pub/sub 消息丢失 | 通知/评论推送不可靠 | 增加消息确认机制，离线用户轮询兜底 |
